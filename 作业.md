        //在数组中这个随机值位置的数和arr[i]位置的数据交换，循环结束，产生随机数组。
        function randSort(arr) {
            for(var i = 0, len = arr.length; i < len; i++) {
                var rand = parseInt(Math.random() * len);
                var temp = arr[rand];
                arr[rand] = arr[i];
                arr[i] = temp;
            }
            return arr;
        }

        console.log(randSort(arr));
